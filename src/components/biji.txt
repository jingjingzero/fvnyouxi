<script setup>
import { ref } from 'vue'
import { gsap } from 'gsap'
import fireballImg from '@/assets/logo/huoqiu.png'
import explosionImg from '@/assets/logo/huiqiubaozha.png'

const container = ref(null)

// ğŸ‘‰ ç§»åŠ¨è·ç¦»å˜é‡
const fireballDistance = 300

const shootFireball = () => {
  const img = document.createElement('img')
  img.src = fireballImg
  img.className = 'absolute top-38 left--25 w-20 h-auto'
  img.style.opacity = '1'

  container.value.appendChild(img)

  gsap.to(img, {
    x: fireballDistance,
    scale: 1.3,
    opacity: 1,
    duration: 1,
     ease: 'linear', // âœ… åŒ€é€ŸåŠ¨ç”»
    onComplete: () => {
      const startLeft = img.offsetLeft
      const startTop = img.offsetTop
      const finalLeft = startLeft + fireballDistance

      img.remove()

      // ğŸ’¥ çˆ†ç‚¸å›¾
      const explosion = document.createElement('img')
      explosion.src = explosionImg
      explosion.className = 'absolute w-30 h-auto pointer-events-none'
      explosion.style.left = `${finalLeft}px`
      explosion.style.top = `${startTop}px`
      explosion.style.opacity = '0'

      container.value.appendChild(explosion)

      gsap.to(explosion, {
        scale: 1.4,
        opacity: 1,
        duration: 0.2,
        ease: 'back.out(2)',
        onComplete: () => {
          gsap.to(explosion, {
            opacity: 0,
            duration: 0.3,
            delay: 0.2,
            onComplete: () => {
              explosion.remove()
            }
          })
        }
      })
    }
  })
}
